// Solution from StackOverflow (answered 2013)
// Accepted answer • 4,729 upvotes • "This saved my life"

function whyIsThisNotWorking(code) {
  try {
    return code.split(' ').reverse().join(' ');
  } catch (e) {
    console.log("idk try turning it off and on");
  }
}

// UPDATE: Still works in 2024. Don't ask why.
